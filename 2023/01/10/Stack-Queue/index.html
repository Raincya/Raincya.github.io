1.栈

代码：

const int N = 1e6 + 10;
int stk[N],tt = 0; // tt 栈顶

void insert(int x){
    stk[++tt] = x;
}

void pop(){
    tt--;
}

int head(){
    return stk[tt];
}

bool empty(){
    return tt > 0; // true 则不空
}

2.队列

(1) 普通队列

代码：

const int N = 1e6 + 10;
int q[N],head = 0,tail = 1;

void insert(int x){
    q[++tail] = x;
}

void pop(){
    head++;
}

int front(){
    return q[head];
}

bool empty(){
    return head <= tail; // true 则不空
}

(2) 循环队列

代码：

const int N = 1e6 + 10;
int q[N],head = 0,tail = 0;

void insert(int x,int n){
    q[tail++] = x;
    if(tail == n) tail = 0; // 队尾是所有待输入元素的最后一个则让其和队首相接
}

void pop(int n){
    head++;
    if(head == n) head = 0;
}

int front(){
    return q[head];
}

bool empty(){
    return head != tail; // true 则为空
}

3.单调栈/队列

(1) 单调栈

代码：

int tt = 0;
for(int i = 1;i <= n;i++){
    while(tt && check(stk[tt],i)) tt--; // check() 确定单调递减||递增
    stk[++tt] = i;
}

(2) 单调队列

代码：

int head = 0,tail = 1;
for(int i = 1;i <= n;i++){
    while(head <= tail && check_out(q[head])) head++; // check_out() 判断是否不满足题意
    while(head <= tail && check(q[tail],i)) tail--; // check() 判断单调递减||递增
    q[++tail] = i;
}
