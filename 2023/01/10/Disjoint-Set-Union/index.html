Disjoint Set Union

2023-01-10

并查集 是一种 树型 的数据结构，用于处理一些 不相交集 的合并及查询问题

主要有如下两种操作：

  1. 合并两个集合（merge）
  2. 判断两个元素是否在同一个集合中（find）

代码：

namespace DSU{
    const int N = 1e6 + 10;
    int fa[N],r[N];
    // 初始化
    void init(int n){
        for(int i = 1;i <= n;i++) 
            fa[i] = i,r[i] = 1;
    }
    // 查找（路径压缩）
    int find(int x){
        if(x == fa[x]) return x;
        else {
            fa[x] = find(fa[x]);
            return fa[x];
        }
    }
    // 合并（按秩合并）
    void merge(int x,int y){
        int fx = find(x),fy = find(y);
        if(r[fx] <= r[fy]) fa[fx] = y;
        else fa[fy] = fx;
        if(r[fx] == r[fy] && fx != fy) r[y]++;
    }
}

用法：

DSU::init(n);
DSU::find(x);
DSU::merge(x,y);
